FastAPI和Streamlit

牛逼的新GitHub，自带全栈后端(fastAPI)和前端(Streamlit)。你可以直接提供这个框架，如果你想跳出flask WSGI框架，深入了解ASGI的方方面面，这是一个不错的选择。

FastAPI和Streamlit都可以通过Docker Compose进行桥接:

version: '3'

services:
  fastapi:
    build: fastapi/
    ports: 
      - 8000:8000
    networks:
      - deploy_network
    container_name: fastapi

  streamlit:
    build: streamlit/
    depends_on:
      - fastapi
    ports: 
        - 8501:8501
    networks:
      - deploy_network
    container_name: streamlit

networks:
  deploy_network:
    driver: bridge
